let i,n;
n=prompt("Input the Size of Array=");
n=parseInt(n);
let nums=[];
nums[0]=0;
nums[1]=1;
for(i=2;i<=n;i++) {
    if(i%2==0) {
       nums[i]=nums[Math.floor(i/2)]; 
    }
    else {
       nums[i]=nums[Math.floor(i/2)]+nums[Math.floor(i/2)+1];
    }
}
console.log("The arrary is",nums);

let max=nums[0];
let k;
for(k=0;k<=nums.length;k++) {
    if(nums[k]>max) {
        max=nums[k];
    }
}
console.log("The Maximum element is=",max)
